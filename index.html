Pregunta: ¿Existe alguna correlación entre el estado civil y el nivel académico de los migrantes colombianos en función de su área de conocimiento y subárea de especialización?

SELECT estado_civil, __rea_Conocimiento, sub_area_conocimiento, Nivel_Acad__mico , COUNT(*) AS cantidad FROM colombiext.colombext WHERE __rea_Conocimiento != 'NO INDICA' AND __rea_Conocimiento != 'NINGUNA' AND sub_area_conocimiento != 'NO INDICA' AND sub_area_conocimiento != 'NINGUNA' AND Nivel_Acad__mico != 'NO INDICA' GROUP BY estado_civil, __rea_Conocimiento, sub_area_conocimiento, Nivel_Acad__mico ORDER BY cantidad DESC;
